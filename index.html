LLMs & Chat Models

from langchain_openai import OpenAI, ChatOpenAI

Chains — LCEL (LangChain Expression Language)
Composable pipeline with | operator
from langchain_core.runnables import (
    RunnablePassthrough,
    RunnableParallel,
    RunnableLambda,
    RunnableSequence
)

# Basic LCEL chain
chain = prompt | llm | StrOutputParser()
result = chain.invoke({"topic": "ML"})

# Parallel execution
parallel = RunnableParallel(
    summary=summarize_chain,
    keywords=keyword_chain
)
parallel.invoke({"text": doc})
# Pass-through (inject context)
chain = (
    RunnablePassthrough.assign(
        context=lambda x: retriever.invoke(x["question"])
    )
    | prompt | llm | StrOutputParser()
)

# Lambda for custom transform
upper = RunnableLambda(lambda x: x.upper())

# Methods on any chain
chain.invoke(input_dict)
chain.batch([dict1, dict2])
async for tok in chain.astream(inp): ...
chain.get_graph().print_ascii()  # visualize

Agents & Tools
ReAct / Tool-calling agents with autonomous decision-making
from langchain.agents import (
    create_react_agent,
    create_tool_calling_agent,
    AgentExecutor
)
from langchain_core.tools import tool
from langchain_community.tools import (
    DuckDuckGoSearchRun,
    WikipediaQueryRun,
    PythonREPLTool
)

# Define a custom tool
@tool
def calculator(expr: str) -> str:
    """Evaluate math expression"""
    return str(eval(expr))
# Build agent
tools = [DuckDuckGoSearchRun(), calculator]
agent = create_tool_calling_agent(llm, tools, prompt)

executor = AgentExecutor(
    agent=agent,
    tools=tools,
    verbose=True,
    max_iterations=10,
    handle_parsing_errors=True
)

result = executor.invoke({"input": "What is 42*9?"})
print(result["output"])

# Stream agent steps
for step in executor.stream({"input": "search AI news"}):
    print(step)

Full RAG Pipeline
Complete Retrieval-Augmented Generation with LCEL
from langchain_openai import ChatOpenAI, OpenAIEmbeddings
from langchain_community.vectorstores import FAISS
from langchain_community.document_loaders import PyPDFLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough

# 1. Load → Split → Embed → Store
docs    = PyPDFLoader("data.pdf").load()
chunks  = RecursiveCharacterTextSplitter(chunk_size=1000, chunk_overlap=200).split_documents(docs)
db      = FAISS.from_documents(chunks, OpenAIEmbeddings())
retriever = db.as_retriever(search_kwargs={"k": 4})

# 2. Prompt
prompt = ChatPromptTemplate.from_template("""
Answer using the context below. If unknown, say "I don't know."
Context: {context}
Question: {question}""")

# 3. Chain
rag_chain = (
    {"context": retriever, "question": RunnablePassthrough()}
    | prompt | ChatOpenAI() | StrOutputParser()
)
answer = rag_chain.invoke("What is the main finding?")

LangGraph Basics
Stateful multi-agent workflows
from langgraph.graph import StateGraph, END
from typing import TypedDict

class State(TypedDict):
    messages: list; next_step: str

def agent_node(state): ...
def tool_node(state):  ...

graph = StateGraph(State)
graph.add_node("agent", agent_node)
graph.add_node("tools", tool_node)
graph.set_entry_point("agent")
graph.add_edge("tools", "agent")
graph.add_conditional_edges("agent",
    lambda s: s["next_step"],
    {"tools": "tools", "done": END})
app = graph.compile()
app.invoke({"messages": ["Start"]})

Few-Shot Prompting
Examples in prompt template
from langchain_core.prompts import FewShotPromptTemplate

examples = [
    {"input": "happy", "output": "sad"},
    {"input": "tall",  "output": "short"}
]
ex_pt = PromptTemplate(
    template="Word: {input} → Antonym: {output}",
    input_variables=["input", "output"])

fs = FewShotPromptTemplate(
    examples=examples,
    example_prompt=ex_pt,
    suffix="Word: {input} → Antonym:",
    input_variables=["input"])

print(fs.format(input="fast"))
